#include <stdio.h>

int main() {
    int n;
    scanf("%d", &n);

    int a[n][n];
    int i, j, temp;

    // Input matrix
    for(i = 0; i < n; i++)
        for(j = 0; j < n; j++)
            scanf("%d", &a[i][j]);

    // Step 1: Transpose
    for(i = 0; i < n; i++) {
        for(j = i + 1; j < n; j++) {
            temp = a[i][j];
            a[i][j] = a[j][i];
            a[j][i] = temp;
        }
    }

    // Step 2: Reverse each row
    for(i = 0; i < n; i++) {
        for(j = 0; j < n / 2; j++) {
            temp = a[i][j];
            a[i][j] = a[i][n - 1 - j];
            a[i][n - 1 - j] = temp;
        }
    }

    // Print rotated matrix
    for(i = 0; i < n; i++) {
        for(j = 0; j < n; j++)
            printf("%d ", a[i][j]);
        printf("\n");
    }

    return 0;
}
