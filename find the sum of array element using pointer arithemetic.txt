#include <stdio.h>

int main()
{
    int n, i, sum = 0;

    scanf("%d", &n);

    int arr[n];
    int *ptr = arr;

    for(i = 0; i < n; i++)
    {
        scanf("%d", ptr + i);
    }

    for(i = 0; i < n; i++)
    {
        sum = sum + *(ptr + i);
    }

    printf("%d", sum);

    return 0;
}
