#include <stdio.h>

int main()
{
    char str[100];
    int freq[256] = {0};
    int i;

    scanf("%s", str);

    // Count frequency
    for(i = 0; str[i] != '\0'; i++)
        freq[str[i]]++;

    // Find first non-repeating character
    for(i = 0; str[i] != '\0'; i++)
    {
        if(freq[str[i]] == 1)
        {
            printf("%c", str[i]);
            return 0;
        }
    }

    printf("No non-repeating character");

    return 0;
}
